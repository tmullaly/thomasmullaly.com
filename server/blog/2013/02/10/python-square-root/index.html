# sqrt by Newton-Raphson
def sqrtNR(x, epsilon):
    x = float(x)
    guess = x/2.0
    diff = guess**2 - x
    count = 0
    while abs(diff) > epsilon:
        count += 1
        print "count =", count, "guess =", guess, "diff =", diff
        guess = guess - diff/(2.0*guess)
        diff = guess**2 - x
    print "number of guesses =", count
    print "guess =", guess
